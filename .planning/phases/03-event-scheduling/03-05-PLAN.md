```kotlin
package ch.teamorg.ui.events

import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import ch.teamorg.domain.*
import ch.teamorg.repository.EventRepository
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
import kotlinx.datetime.*

data class EventListState(
    val events: List<EventWithTeams> = emptyList(),
    val isLoading: Boolean = false,
    val error: String? = null,
    val selectedTeamId: String? = null,   // null = "All Teams"
    val selectedType: String? = null,     // null = "All"
    val isCoach: Boolean = false,
    val teams: List<MatchedTeam> = emptyList()  // user's teams for filter chips
)

class EventListViewModel(
    private val eventRepository: EventRepository
) : ViewModel() {
    private val _state = MutableStateFlow(EventListState())
    val state = _state.asStateFlow()

    private val _events = MutableSharedFlow<EventListEvent>()
    val events = _events.asSharedFlow()

    init { loadEvents() }

    fun loadEvents() {
        viewModelScope.launch {
            _state.update { it.copy(isLoading = true, error = null) }
            eventRepository.getMyEvents(
                type = _state.value.selectedType,
                teamId = _state.value.selectedTeamId
            ).onSuccess { events ->
                _state.update { it.copy(events = events, isLoading = false) }
                // Extract unique teams from events for filter chips
                val teams = events.flatMap { it.matchedTeams }.distinctBy { it.id }
                _state.update { it.copy(teams = teams) }
            }.onFailure { e ->
                _state.update { it.copy(error = e.message, isLoading = false) }
            }
        }
    }

    fun setTeamFilter(teamId: String?) {
        _state.update { it.copy(selectedTeamId = teamId) }
        loadEvents()
    }

    fun setTypeFilter(type: String?) {
        _state.update { it.copy(selectedType = type) }
        loadEvents()
    }
}

sealed class EventListEvent {
    data class NavigateToDetail(val eventId: String) : EventListEvent()
    data object NavigateToCreate : EventListEvent()
}
```

```kotlin
@Composable
fun EventListScreen(
    viewModel: EventListViewModel,
    onEventClick: (String) -> Unit,
    onCreateClick: () -> Unit
) {
    val state by viewModel.state.collectAsState()

    Scaffold(
        topBar = {
            TopAppBar(title = { Text("Events") })
        },
        floatingActionButton = {
            if (state.isCoach) {
                FloatingActionButton(onClick = onCreateClick) {
                    Icon(Icons.Default.Add, contentDescription = "Create event")
                }
            }
        }
    ) { padding ->
        Column(modifier = Modifier.padding(padding)) {
            // Filter row (horizontal scroll)
            FilterChipRow(
                teams = state.teams,
                selectedTeamId = state.selectedTeamId,
                selectedType = state.selectedType,
                onTeamSelected = viewModel::setTeamFilter,
                onTypeSelected = viewModel::setTypeFilter
            )

            if (state.isLoading) {
                CircularProgressIndicator(modifier = Modifier.align(Alignment.CenterHorizontally))
            } else if (state.events.isEmpty()) {
                // Empty state
                EmptyEventsList(isCoach = state.isCoach, onCreateClick = onCreateClick)
            } else {
                LazyColumn {
                    items(state.events, key = { it.event.id }) { ewt ->
                        EventListItem(
                            event = ewt.event,
                            matchedTeams = ewt.matchedTeams,
                            onClick = { onEventClick(ewt.event.id) }
                        )
                    }
                }
            }
        }
    }
}
```

Create `EventDetailViewModel.kt`:

```kotlin
data class EventDetailState(
    val event: EventWithTeams? = null,
    val isLoading: Boolean = false,
    val error: String? = null,
    val isCoach: Boolean = false,
    val refreshVersion: Int = 0  // incremented to trigger re-fetch after edit/cancel
)

class EventDetailViewModel(
    private val eventRepository: EventRepository
) : ViewModel() {
    private val _state = MutableStateFlow(EventDetailState())
    val state = _state.asStateFlow()

    fun loadEvent(eventId: String) {
        viewModelScope.launch {
            _state.update { it.copy(isLoading = true) }
            eventRepository.getEventDetail(eventId)
                .onSuccess { ewt -> _state.update { it.copy(event = ewt, isLoading = false) } }
                .onFailure { e -> _state.update { it.copy(error = e.message, isLoading = false) } }
        }
    }

    /** Call after returning from edit/cancel to trigger LaunchedEffect re-fetch. */
    fun requestRefresh() {
        _state.update { it.copy(refreshVersion = it.refreshVersion + 1) }
    }
}
```

IMPORTANT -- Stale data fix: `LaunchedEffect(screen.eventId)` alone will NOT re-fire when navigating back from edit (same eventId). The ViewModel exposes `refreshVersion` as a counter. In AppNavigation, the LaunchedEffect keys on BOTH `screen.eventId` AND `state.refreshVersion`, so incrementing `refreshVersion` via `requestRefresh()` triggers a re-fetch. The `onEdit` callback in AppNavigation must call `viewModel.requestRefresh()` after the edit screen pops (see navigation wiring below).
