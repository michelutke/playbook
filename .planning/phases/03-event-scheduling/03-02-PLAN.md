```kotlin
package ch.teamorg.routes

import ch.teamorg.domain.models.*
import ch.teamorg.domain.repositories.EventRepository
import io.ktor.http.*
import io.ktor.server.auth.*
import io.ktor.server.auth.jwt.*
import io.ktor.server.request.*
import io.ktor.server.response.*
import io.ktor.server.routing.*
import org.koin.ktor.ext.get as koinGet
import java.time.Instant
import java.util.UUID

fun Routing.eventRoutes() {
    val eventRepository = koinGet<EventRepository>()

    authenticate("jwt") {
        // GET /users/me/events -- player's deduplicated event list
        get("/users/me/events") {
            val userId = call.principal<JWTPrincipal>()!!.payload.getClaim("userId").asString().let { UUID.fromString(it) }
            val from = call.parameters["from"]?.let { Instant.parse(it) }
            val to = call.parameters["to"]?.let { Instant.parse(it) }
            val type = call.parameters["type"]
            val teamId = call.parameters["teamId"]?.let { UUID.fromString(it) }
            val events = eventRepository.findEventsForUser(userId, from, to, type, teamId)
            call.respond(events)
        }

        // GET /teams/{teamId}/events -- team-scoped event list
        get("/teams/{teamId}/events") {
            val teamId = UUID.fromString(call.parameters["teamId"])
            val from = call.parameters["from"]?.let { Instant.parse(it) }
            val to = call.parameters["to"]?.let { Instant.parse(it) }
            val events = eventRepository.findEventsForTeam(teamId, from, to)
            call.respond(events)
        }

        // GET /events/{id} -- event detail with teams
        get("/events/{id}") {
            val id = UUID.fromString(call.parameters["id"])
            val event = eventRepository.findByIdWithTeams(id)
            if (event != null) call.respond(event)
            else call.respond(HttpStatusCode.NotFound)
        }

        // POST /events -- create event (one-off or series)
        post("/events") {
            val userId = call.principal<JWTPrincipal>()!!.payload.getClaim("userId").asString().let { UUID.fromString(it) }
            val request = call.receive<CreateEventRequest>()
            // TODO Phase 5: emit domain event for notifications
            if (request.recurring != null) {
                val series = eventRepository.createSeries(request, userId)
                eventRepository.materialiseUpcomingOccurrences()
                val events = eventRepository.findEventsForUser(userId, from = request.startAt, to = null, type = null, teamId = null)
                call.respond(HttpStatusCode.Created, events.firstOrNull() ?: mapOf("seriesId" to series.id))
            } else {
                val event = eventRepository.create(request, userId)
                call.respond(HttpStatusCode.Created, event)
            }
        }

        // PATCH /events/{id} -- edit event (with scope for recurring)
        //
        // Deserialization approach: The client sends EditEventRequest which has an
        // optional `scope` field ("this_only" | "this_and_future" | "all").
        // The server-side EditEventRequest (Plan 01 Event.kt) does NOT have scope.
        // Therefore: create a server-side wrapper data class to receive both:
        //
        //   @Serializable
        //   data class EditEventWithScope(
        //       val scope: String? = "this_only",
        //       val title: String? = null,
        //       val type: String? = null,
        //       val startAt: Instant? = null,
        //       val endAt: Instant? = null,
        //       val meetupAt: Instant? = null,
        //       val location: String? = null,
        //       val description: String? = null,
        //       val minAttendees: Int? = null,
        //       val teamIds: List<UUID>? = null,
        //       val subgroupIds: List<UUID>? = null
        //   )
        //
        // Define this class in EventRoutes.kt (private, route-local).
        // Then convert to EditEventRequest for the repository call.
        patch("/events/{id}") {
            val id = UUID.fromString(call.parameters["id"])
            val body = call.receive<EditEventWithScope>()
            val scope = RecurringScope.valueOf(body.scope ?: "this_only")
            val editRequest = EditEventRequest(
                title = body.title, type = body.type,
                startAt = body.startAt, endAt = body.endAt,
                meetupAt = body.meetupAt, location = body.location,
                description = body.description, minAttendees = body.minAttendees,
                teamIds = body.teamIds, subgroupIds = body.subgroupIds
            )

            val existing = eventRepository.findById(id)
            if (existing == null) { call.respond(HttpStatusCode.NotFound); return@patch }

            when (scope) {
                RecurringScope.this_only -> {
                    eventRepository.update(id, editRequest)
                    // If part of series, mark seriesOverride = true
                }
                RecurringScope.this_and_future -> {
                    // Split series: end current series at this sequence, create new series with updated template
                    if (existing.seriesId != null && existing.seriesSequence != null) {
                        eventRepository.cancelFutureInSeries(existing.seriesId, existing.seriesSequence)
                        // Create new series with updated fields, materialise future
                    }
                }
                RecurringScope.all -> {
                    if (existing.seriesId != null) {
                        eventRepository.updateSeriesTemplate(existing.seriesId, editRequest)
                        eventRepository.updateFutureInSeries(existing.seriesId, existing.seriesSequence ?: 0, editRequest)
                    } else {
                        eventRepository.update(id, editRequest)
                    }
                }
            }
            // TODO Phase 5: emit domain event for notifications
            val updated = eventRepository.findByIdWithTeams(id)
            call.respond(updated ?: HttpStatusCode.NotFound)
        }

        // POST /events/{id}/cancel -- cancel event (with scope for recurring)
        post("/events/{id}/cancel") {
            val id = UUID.fromString(call.parameters["id"])
            val body = call.receive<Map<String, Any?>>()
            val scopeStr = (body["scope"] as? String) ?: "this_only"
            val scope = RecurringScope.valueOf(scopeStr)

            val existing = eventRepository.findById(id)
            if (existing == null) { call.respond(HttpStatusCode.NotFound); return@post }

            when (scope) {
                RecurringScope.this_only -> eventRepository.cancel(id)
                RecurringScope.this_and_future -> {
                    if (existing.seriesId != null && existing.seriesSequence != null) {
                        eventRepository.cancel(id)
                        eventRepository.cancelFutureInSeries(existing.seriesId, existing.seriesSequence + 1)
                    } else {
                        eventRepository.cancel(id)
                    }
                }
                RecurringScope.all -> {
                    if (existing.seriesId != null) {
                        eventRepository.cancelFutureInSeries(existing.seriesId, 0)
                    }
                    eventRepository.cancel(id)
                }
            }
            // TODO Phase 5: emit domain event for notifications
            call.respond(HttpStatusCode.OK)
        }

        // POST /events/{id}/duplicate -- clone event
        post("/events/{id}/duplicate") {
            val userId = call.principal<JWTPrincipal>()!!.payload.getClaim("userId").asString().let { UUID.fromString(it) }
            val id = UUID.fromString(call.parameters["id"])
            val event = eventRepository.duplicate(id, userId)
            if (event != null) call.respond(HttpStatusCode.Created, event)
            else call.respond(HttpStatusCode.NotFound)
        }
    }
}
```

```kotlin
package ch.teamorg.infra

import ch.teamorg.domain.repositories.EventRepository
import io.ktor.server.application.*
import kotlinx.coroutines.*
import org.koin.ktor.ext.get as koinGet
import org.slf4j.LoggerFactory
import kotlin.time.Duration.Companion.hours

private val logger = LoggerFactory.getLogger("EventMaterialisationJob")

fun Application.startMaterialisationJob() {
    val eventRepository = koinGet<EventRepository>()

    launch(Dispatchers.IO) {
        // Initial run on startup
        try {
            val count = eventRepository.materialiseUpcomingOccurrences()
            logger.info("Initial materialisation complete: $count new occurrences created")
        } catch (e: Exception) {
            logger.error("Initial materialisation failed", e)
        }

        // Daily recurring run
        while (isActive) {
            delay(24.hours)
            try {
                val count = eventRepository.materialiseUpcomingOccurrences()
                logger.info("Daily materialisation complete: $count new occurrences created")
            } catch (e: Exception) {
                logger.error("Daily materialisation failed", e)
            }
        }
    }
}
```
