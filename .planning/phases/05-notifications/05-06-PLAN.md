    ```kotlin
    class PushServiceTest {
        @Test
        fun `sendToUsers calls OneSignal REST API with correct payload`() {
            var capturedRequest: HttpRequestData? = null
            val mockEngine = MockEngine { request ->
                capturedRequest = request
                respond(content = """{"id":"abc"}""", status = HttpStatusCode.OK,
                    headers = headersOf(HttpHeaders.ContentType, "application/json"))
            }
            val client = HttpClient(mockEngine) {
                install(ContentNegotiation) { json() }
            }
            val pushService = PushServiceImpl(client)

            runBlocking {
                pushService.sendToUsers(
                    userIds = listOf("user-1", "user-2"),
                    title = "Test Title",
                    body = "Test Body",
                    data = mapOf("entity_id" to "event-123")
                )
            }

            assertNotNull(capturedRequest)
            assertEquals("https://api.onesignal.com/notifications?c=push", capturedRequest!!.url.toString())
            val bodyText = (capturedRequest!!.body as TextContent).text
            assertTrue(bodyText.contains("include_aliases"))
            assertTrue(bodyText.contains("user-1"))
            assertTrue(bodyText.contains("user-2"))
            assertTrue(bodyText.contains("Test Title"))
            assertTrue(bodyText.contains("Test Body"))
        }

        @Test
        fun `sendToUsers with empty list does not call API`() {
            var called = false
            val mockEngine = MockEngine { called = true; respondOk() }
            val client = HttpClient(mockEngine)
            val pushService = PushServiceImpl(client)

            runBlocking { pushService.sendToUsers(emptyList(), "t", "b") }
            assertFalse(called)
        }

        @Test
        fun `sendToUsers handles API error gracefully`() {
            val mockEngine = MockEngine {
                respond("", status = HttpStatusCode.InternalServerError)
            }
            val client = HttpClient(mockEngine) {
                install(ContentNegotiation) { json() }
            }
            val pushService = PushServiceImpl(client)

            // Should not throw
            runBlocking {
                pushService.sendToUsers(listOf("user-1"), "t", "b")
            }
        }
    }
    ```
