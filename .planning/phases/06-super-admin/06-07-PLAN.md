private suspend fun ApplicationTestBuilder.registerSuperAdmin(client: HttpClient): AuthResponse {
    val auth = client.post("/auth/register") {
        contentType(ContentType.Application.Json)
        setBody(RegisterRequest("superadmin@test.com", "password123", "Super Admin"))
    }.body<AuthResponse>()

    // Directly set isSuperAdmin in DB (no API for this — manual only per AUTH-06)
    transaction {
        UsersTable.update({ UsersTable.id eq UUID.fromString(auth.userId) }) {
            it[isSuperAdmin] = true
        }
    }
    return auth
}
