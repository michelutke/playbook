```ts
import { apiGet } from '$lib/server/api';

export const load = async ({ locals, url }) => {
  const query = url.searchParams.get('q') || '';
  const page = parseInt(url.searchParams.get('page') || '1');

  // Only search if query has at least 2 chars (per UI spec)
  if (query.length >= 2) {
    const users = await apiGet(`/admin/users?q=${encodeURIComponent(query)}&page=${page}&pageSize=50`, locals.token!);
    return { users, query };
  }

  return { users: { users: [], page: 1, pageSize: 50, totalCount: 0 }, query };
};
```

**admin/src/routes/admin/audit-log/+page.server.ts**:

```ts
import { apiGet } from '$lib/server/api';

export const load = async ({ locals, url }) => {
  const action = url.searchParams.get('action') || undefined;
  const actor = url.searchParams.get('actor') || undefined;
  const startDate = url.searchParams.get('startDate') || undefined;
  const endDate = url.searchParams.get('endDate') || undefined;
  const page = parseInt(url.searchParams.get('page') || '1');

  const params = new URLSearchParams();
  params.set('page', String(page));
  params.set('pageSize', '50');
  if (action) params.set('action', action);
  if (actor) params.set('actor', actor);
  if (startDate) params.set('startDate', startDate);
  if (endDate) params.set('endDate', endDate);

  const log = await apiGet(`/admin/audit-log?${params.toString()}`, locals.token!);
  return { log, filters: { action, actor, startDate, endDate } };
};
```
