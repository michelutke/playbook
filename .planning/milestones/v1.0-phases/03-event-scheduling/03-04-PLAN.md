```sql
CREATE TABLE IF NOT EXISTS CachedEvent (
    id TEXT NOT NULL PRIMARY KEY,
    title TEXT NOT NULL,
    type TEXT NOT NULL,
    start_at INTEGER NOT NULL,
    end_at INTEGER NOT NULL,
    meetup_at INTEGER,
    location TEXT,
    description TEXT,
    min_attendees INTEGER,
    status TEXT NOT NULL DEFAULT 'active',
    cancelled_at INTEGER,
    series_id TEXT,
    series_sequence INTEGER,
    series_override INTEGER NOT NULL DEFAULT 0,
    created_by TEXT NOT NULL,
    team_ids TEXT NOT NULL DEFAULT '',
    subgroup_ids TEXT NOT NULL DEFAULT '',
    matched_teams_json TEXT NOT NULL DEFAULT '[]',
    cached_at INTEGER NOT NULL
);

getUpcomingEvents:
SELECT * FROM CachedEvent
WHERE start_at >= :fromMillis AND start_at <= :toMillis
ORDER BY start_at ASC;

getEventById:
SELECT * FROM CachedEvent
WHERE id = :eventId;

getFilteredEvents:
SELECT * FROM CachedEvent
WHERE start_at >= :fromMillis AND start_at <= :toMillis
AND (CASE WHEN :type IS NOT NULL THEN type = :type ELSE 1 END)
AND (CASE WHEN :teamId IS NOT NULL THEN team_ids LIKE '%' || :teamId || '%' ELSE 1 END)
ORDER BY start_at ASC;

upsertEvent:
INSERT OR REPLACE INTO CachedEvent(id, title, type, start_at, end_at, meetup_at, location, description, min_attendees, status, cancelled_at, series_id, series_sequence, series_override, created_by, team_ids, subgroup_ids, matched_teams_json, cached_at)
VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?);

deleteOlderThan:
DELETE FROM CachedEvent WHERE cached_at < :cutoffMillis;

deleteAll:
DELETE FROM CachedEvent;
```

```kotlin
package ch.teamorg.data

import ch.teamorg.domain.*
import kotlinx.datetime.*
import kotlinx.serialization.json.Json
import kotlinx.serialization.encodeToString
import kotlinx.serialization.decodeFromString

class EventCacheManager(private val database: /* SQLDelight database type */) {

    private val queries = database.eventQueries  // generated by SQLDelight

    fun saveEvents(events: List<EventWithTeams>) {
        val now = Clock.System.now().toEpochMilliseconds()
        events.forEach { ewt ->
            val e = ewt.event
            queries.upsertEvent(
                id = e.id,
                title = e.title,
                type = e.type,
                start_at = e.startAt.toEpochMilliseconds(),
                end_at = e.endAt.toEpochMilliseconds(),
                meetup_at = e.meetupAt?.toEpochMilliseconds(),
                location = e.location,
                description = e.description,
                min_attendees = e.minAttendees?.toLong(),
                status = e.status,
                cancelled_at = e.cancelledAt?.toEpochMilliseconds(),
                series_id = e.seriesId,
                series_sequence = e.seriesSequence?.toLong(),
                series_override = if (e.seriesOverride) 1L else 0L,
                created_by = e.createdBy,
                team_ids = e.teamIds.joinToString(","),
                subgroup_ids = e.subgroupIds.joinToString(","),
                matched_teams_json = Json.encodeToString(ewt.matchedTeams),
                cached_at = now
            )
        }
    }

    fun getOfflineEvents(fromMillis: Long, toMillis: Long): List<EventWithTeams> {
        return queries.getUpcomingEvents(fromMillis, toMillis).executeAsList().map { row ->
            rowToEventWithTeams(row)
        }
    }

    fun getFilteredOfflineEvents(fromMillis: Long, toMillis: Long, type: String?, teamId: String?): List<EventWithTeams> {
        return queries.getFilteredEvents(fromMillis, toMillis, type, teamId).executeAsList().map { row ->
            rowToEventWithTeams(row)
        }
    }

    fun cleanup() {
        // Remove events cached more than 7 days ago
        val cutoff = Clock.System.now().minus(7, DateTimeUnit.DAY, TimeZone.UTC).toEpochMilliseconds()
        queries.deleteOlderThan(cutoff)
    }

    // Helper to map SQLDelight row back to domain model
    // (adapt column names to match generated code)
}
```

Create `EventRepositoryImpl.kt`:

```kotlin
package ch.teamorg.data.repository

import ch.teamorg.domain.*
import ch.teamorg.repository.EventRepository
import io.ktor.client.*
import io.ktor.client.call.*
import io.ktor.client.request.*
import io.ktor.http.*
import kotlinx.datetime.*

class EventRepositoryImpl(
    private val httpClient: HttpClient,
    private val cacheManager: EventCacheManager
) : EventRepository {

    override suspend fun getMyEvents(from: String?, to: String?, type: String?, teamId: String?): Result<List<EventWithTeams>> {
        return try {
            val events: List<EventWithTeams> = httpClient.get("/users/me/events") {
                from?.let { parameter("from", it) }
                to?.let { parameter("to", it) }
                type?.let { parameter("type", it) }
                teamId?.let { parameter("teamId", it) }
            }.body()
            // Cache for offline access
            cacheManager.saveEvents(events)
            Result.success(events)
        } catch (e: io.ktor.client.network.sockets.ConnectTimeoutException) {
            // Network timeout -- fall back to cache
            offlineFallback(type, teamId)
        } catch (e: io.ktor.utils.io.errors.IOException) {
            // No connectivity -- fall back to cache
            offlineFallback(type, teamId)
        } catch (e: io.ktor.client.plugins.HttpRequestTimeoutException) {
            // Request timeout -- fall back to cache
            offlineFallback(type, teamId)
        }
        // All other exceptions (including ResponseException for 4xx/5xx)
        // propagate to the caller -- per CLAUDE.md "catch specific exceptions" rule
    }

    private fun offlineFallback(type: String?, teamId: String?): Result<List<EventWithTeams>> {
        val now = Clock.System.now().toEpochMilliseconds()
        val threeMonths = Clock.System.now().plus(90, DateTimeUnit.DAY, TimeZone.UTC).toEpochMilliseconds()
        return Result.success(cacheManager.getFilteredOfflineEvents(now, threeMonths, type, teamId))
    }

    override suspend fun getEventDetail(id: String): Result<EventWithTeams> = runCatching {
        httpClient.get("/events/$id").body()
    }

    override suspend fun createEvent(request: CreateEventRequest): Result<Event> = runCatching {
        httpClient.post("/events") {
            contentType(ContentType.Application.Json)
            setBody(request)
        }.body()
    }

    override suspend fun editEvent(id: String, request: EditEventRequest): Result<Event> = runCatching {
        httpClient.patch("/events/$id") {
            contentType(ContentType.Application.Json)
            setBody(request)
        }.body()
    }

    override suspend fun cancelEvent(id: String, scope: String): Result<Unit> = runCatching {
        httpClient.post("/events/$id/cancel") {
            contentType(ContentType.Application.Json)
            setBody(mapOf("scope" to scope))
        }
        Unit
    }

    override suspend fun duplicateEvent(id: String): Result<Event> = runCatching {
        httpClient.post("/events/$id/duplicate").body()
    }

    override suspend fun getSubGroups(teamId: String): Result<List<SubGroup>> = runCatching {
        httpClient.get("/teams/$teamId/subgroups").body()
    }
}
```
